class AnimalShelf {

    // 总的
    LinkedList<int[]> all;
    // 猫的 0
    LinkedList<int[]> catQ;
    // 狗的 1
    LinkedList<int[]> dogQ;

    public AnimalShelf() {
        all = new LinkedList<>();
        catQ = new LinkedList<>();
        dogQ = new LinkedList<>();
    }

    public void enqueue(int[] animal) {
        all.addLast(animal);
        if (isCat(animal)){
            catQ.addLast(animal);
        }else {
            dogQ.addLast(animal);
        }
    }

    public int[] dequeueAny() {
        if (all.size()==0){
            return new int[]{-1,-1};
        }
        int[] animal = all.removeFirst();
        if (isCat(animal)){
            catQ.removeFirst();
        }else {
            dogQ.removeFirst();
        }
        return animal;
    }

    public int[] dequeueDog() {
        if (dogQ.size()==0){
            return new int[]{-1,-1};
        }
        int[] animal = dogQ.removeFirst();
        all.remove(animal);
        return animal;
    }

    public int[] dequeueCat() {
        if (catQ.size()==0){
            return new int[]{-1,-1};
        }
        int[] animal = catQ.removeFirst();
        all.remove(animal);
        return animal;
    }

    public boolean isCat(int[] animal){
        if (animal[1]==0){
            return true;
        }else {
            return false;
        }
    }
}
